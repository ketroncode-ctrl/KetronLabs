# Ketron Code Editor — AI-powered, lightweight IDE
from ketron import Editor, AIAssist, Collab
editor = Editor(
theme="midnight_dark",
ai=AIAssist(mode="smart"), # catches errors live
collab=Collab(network="local"),
languages=["python", "html", "css", "js"]
)
editor.run() # 🚀 Start coding!
